# Helper function to create or update registry values reliably
function Set-RegistryValue {
    param(
        [Parameter(Mandatory = $true)][string]$Path,
        [Parameter(Mandatory = $true)][string]$Name,
        [Parameter(Mandatory = $true)]$Value,
        [Microsoft.Win32.RegistryValueKind]$Type = [Microsoft.Win32.RegistryValueKind]::DWord
    )

    try {
        if (-not (Test-Path -Path $Path)) {
            New-Item -Path $Path -Force | Out-Null
        }

        $propertyType = switch ($Type) {
            ([Microsoft.Win32.RegistryValueKind]::String) { 'String' }
            ([Microsoft.Win32.RegistryValueKind]::MultiString) { 'MultiString' }
            ([Microsoft.Win32.RegistryValueKind]::Binary) { 'Binary' }
            ([Microsoft.Win32.RegistryValueKind]::QWord) { 'QWord' }
            default { 'DWord' }
        }

        if ((Get-ItemProperty -Path $Path -Name $Name -ErrorAction SilentlyContinue) -eq $null) {
            New-ItemProperty -Path $Path -Name $Name -Value $Value -PropertyType $propertyType -Force | Out-Null
        } else {
            Set-ItemProperty -Path $Path -Name $Name -Value $Value -Force
        }

        $current = (Get-ItemProperty -Path $Path -Name $Name -ErrorAction Stop).$Name
        $expected = switch ($Type) {
            ([Microsoft.Win32.RegistryValueKind]::DWord) { [int]$Value }
            ([Microsoft.Win32.RegistryValueKind]::QWord) { [long]$Value }
            default { $Value }
        }

        if ($current -ne $expected) {
            throw "Verification failed. Expected $expected but found $current."
        }
    } catch {
        Write-Host "ERROR: Failed to set registry value $Path\\$Name - $_" -ForegroundColor Red
        throw
    }
}

# Function to validate URL - ensures the URL is reachable before download
function Test-UrlIsValid {
    param (
        [string]$Url
    )

    try {
        # Simple HEAD request to check if URL is reachable
        $request = [System.Net.WebRequest]::Create($Url)
        $request.Method = "HEAD"
        $request.Timeout = 15000  # 15 seconds timeout
        $request.UserAgent = "Mozilla/5.0 Windows PowerShell Script"

        # Get the response
        $response = $request.GetResponse()

        # Check if we can access the URL
        $success = $response.StatusCode -eq [System.Net.HttpStatusCode]::OK

        # Show file size if available
        $contentLength = $response.Headers["Content-Length"]
        if ($contentLength) {
            $sizeInMB = [math]::Round([long]$contentLength / 1MB, 2)
            if ($sizeInMB -gt 1000) {
                Write-Host "File size: $([math]::Round($sizeInMB / 1024, 2)) GB" -ForegroundColor Cyan
            } else {
                Write-Host "File size: $sizeInMB MB" -ForegroundColor Cyan
            }
        }

        # Close the response
        $response.Close()
        return $success
    }
    catch {
        # Simple error message without details that could cause red dumps
        Write-Host "ERROR: Could not access the URL. Please verify it's correct and accessible." -ForegroundColor Red
        return $false
    }
}

# Function to wait for SetupHost.exe to appear (the definitive sign that upgrade is working)
function Wait-ForSetupHost {

        [int]$TimeoutSeconds = 300,  # Wait up to 5 minutes by default
        [int]$CheckIntervalSeconds = 5
    )

    Write-ProgressLog "Waiting for SetupHost.exe to start (the critical process for Windows upgrades)..."

    $stopwatch = [System.Diagnostics.Stopwatch]::StartNew()
    $timeoutMs = $TimeoutSeconds * 1000

    while ($stopwatch.ElapsedMilliseconds -lt $timeoutMs) {
        $setupHost = Get-Process -Name "SetupHost" -ErrorAction SilentlyContinue
        if ($setupHost) {
            $stopwatch.Stop()
            Write-ProgressLog "SUCCESS: SetupHost.exe started after $([math]::Round($stopwatch.ElapsedMilliseconds / 1000)) seconds."
            Write-ProgressLog "SetupHost.exe PID: $($setupHost.Id), Started at: $(Get-Date)"
            return $true
        }

        $setupPrep = Get-Process -Name "setupprep" -ErrorAction SilentlyContinue
        if ($setupPrep) {
            Write-ProgressLog "setupprep.exe is running (PID: $($setupPrep.Id)). Waiting for it to launch SetupHost.exe..."
        }

        # Sleep before checking again
        Start-Sleep -Seconds $CheckIntervalSeconds
    }

    $stopwatch.Stop()
    Write-ProgressLog "WARNING: SetupHost.exe did not start within $TimeoutSeconds seconds."
    return $false
}

# Launch setup.exe through a SYSTEM scheduled task to guarantee firmware/BCD access
function Invoke-SetupAsSystem {
    param (
        [Parameter(Mandatory = $true)][string]$Executable,
        [Parameter(Mandatory = $true)][string[]]$Arguments,
        [string]$TaskName = 'Win11_24H2 Forced Upgrade'
    )

    $startTime = (Get-Date).AddMinutes(1)
    $startTimeString = $startTime.ToString('HH:mm')
    $startDateString = $startTime.ToString('MM/dd/yyyy')

    $escapedArguments = $Arguments | ForEach-Object {
        $arg = $_ -replace '"', '""'
        if ($arg -match '\s') {
            '"' + $arg + '"'
        } else {
            $arg
        }
    }

    $argumentString = ($escapedArguments -join ' ')
    $commandLine = "cmd.exe /c `"$Executable`" $argumentString"

    try {
        schtasks.exe /End /TN $TaskName /F > $null 2>&1
    } catch {
        # Ignore if the task is not currently running
    }

    try {
        schtasks.exe /Delete /TN $TaskName /F > $null 2>&1
    } catch {
        # Ignore if the task does not exist
    }

    $createArgs = @(
        '/Create','/F','/SC','ONCE','/TN',$TaskName,
        '/TR',$commandLine,
        '/ST',$startTimeString,
        '/SD',$startDateString,
        '/RL','HIGHEST','/RU','SYSTEM'
    )

    $createProcess = Start-Process -FilePath 'schtasks.exe' -ArgumentList $createArgs -Wait -NoNewWindow -PassThru
    if ($createProcess.ExitCode -ne 0) {
        throw "Failed to create SYSTEM scheduled task for setup.exe (exit code $($createProcess.ExitCode))."
    }

    $runArgs = @('/Run','/TN',$TaskName)
    $runProcess = Start-Process -FilePath 'schtasks.exe' -ArgumentList $runArgs -Wait -NoNewWindow -PassThru
    if ($runProcess.ExitCode -ne 0) {
        throw "Failed to launch setup.exe via scheduled task (exit code $($runProcess.ExitCode))."
    }

    Start-Sleep -Seconds 5

    return @{ TaskName = $TaskName; CommandLine = $commandLine }
}
